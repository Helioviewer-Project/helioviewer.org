 12 	 * @constructs
 13 	 * @description Creates a new EventMarker
 14 	 * @param {Object} eventLayer EventLayer associated with the EventMarker
 15 	 * @param {JSON} event Event details
 16 	 * @param {Date} date The date when the given event occured
 17 	 * @param {Int} utcOffset The UTC offset for the system's local time 
 18 	 * @param {Int} sunRadius The radius of the sun in pixels
 19 	 * @param {Object} options Extra EventMarker settings to use
 20 	 */    
 21     initialize: function (eventLayer, event, date, utcOffset, sunRadius, options) {
 22     	Object.extend(this, event);
 23 		Object.extend(this, options);
 24 		
 25 		this.eventLayer = eventLayer;
 26 		this.appDate = date;
 27 		this.utcOffset = utcOffset;
 28 		this.sunRadius = sunRadius;
 29 		
 30 		//Determine event type and catalog
 31 		var catalogs = eventLayer.eventAccordion.eventCatalogs;
 32 		this.catalogName = catalogs.get(this.catalogId).name;
 33 		this.type = catalogs.get(this.catalogId).eventType;
 34 		
 35 		//Create container
 36 		this.pos = {
 37 			x: this.sunX * sunRadius,
 38 			y: this.sunY * sunRadius
 39 		}
 40 		
 41 		this.container = this.eventLayer.domNode.appendChild(
 42 			new Element('div', {className: 'event', style: 'left: ' + (this.pos.x) + 'px; top: ' + (this.pos.y) + 'px;'})
 43 		);
 44 				
 45 		//Create dom-nodes for event marker, details label, and details popup
 46 		this.createMarker();
 47 		this.createLabel();
 48 		this.createPopup();
 49 	},

 51 	/**
 52 	 * @description Creates the marker and adds it to the viewport
 53 	 */
 54 	createMarker: function () {
 55 		//make event-type CSS-friendly
 56 		var cssType = this.type.gsub(' ', "_");
 57 		
 58 		//create html dom-node
 59 		var marker = new Element('div', {className: 'event-marker'});
 60 		marker.setStyle({
 61 			'background': 'url(images/events/' + this.eventLayer.icon + "-" + cssType + '.png)'
 62 		});
 63 		
 64 		//var self = this;
 65 		//marker.observe('click', function(event) {
 66 		//	self.togglePopup();
 67 		//});
 68 		
 69 		this.marker = marker;
 70 		
 71 		this.container.appendChild(marker);
 72 	},

 74 	/**
 75 	 * @description Creates a small block of text which is displayed when the user pressed the "d" key ("details").
 76 	 */
 77 	createLabel: function () {
 78 		var display = (this.eventLayer.displayLabels ? "inline" : "none");
 79 		var labelText = null;
 80 		
 81 		//Determine what to use for label text
 82 		switch (this.type) {
 83 			case "Active Region":
 84 				labelText = this.eventId;
 85 				break;
 86 			case "CME":
 87 				labelText = this.time.startTime;
 88 				break;
 89 			case "Type II Radio Burst":
 90 				labelText = this.time.startTime;
 91 				break;
 92 			default:
 93 				labelText = this.time.startTime;
 94 				break;
 95 		}
 96 		
 97 		//Determine time difference between desired time and event time
 98 		var eventDate = Date.parse(this.time.startTime.substr(0,19)).addSeconds(this.utcOffset);
 99 		var timeDiff = eventDate.getTime() - this.appDate.getTime();
100 		
101 		//Create a hidden node with the events ID to be displayed upon user request
102 		var label = new Element('div', {className: 'event-label'}).setStyle({'display': display}).insert(labelText);
103 		
104 		//Adjust style to reflect time difference
105         if (timeDiff < 0) {
106         	label.addClassName("timeBehind");
107         }
108         else if (timeDiff > 0) {
109         	label.addClassName("timeAhead");
110         }
111         
112         this.label = label;
113         
114         this.container.appendChild(label);
115 		
116 	},

149 	/**
150 	 * @description Creates a popup which is displayed when the event marker is clicked
151 	 */
152 	createPopup: function () {
153 		var properties = $H(this.properties);
154 		var content = "<div class='event-popup-container'>" + 
155 					"<strong>" + this.type + " " + this.eventId + "</strong><br>" +
156 					"<p>" + this.catalogName + "</p><br>" +
157 					"<strong>start:</strong> " + this.time.startTime + "<br>" +
158 					"<strong>end:</strong> " + this.time.endTime + "<br><br>";
159 					
160 					//"<strong>Position Angle:</strong> " + this.polarCpa + "°       " +
161 					//"<strong>Width:</strong> " + this.polarWidth + "°<br>";
162 					
163 			properties.keys().each(function(key){
164 				content += "<strong>" + key + ":</strong> " + properties.get(key);
165 				if (key.search(/angle/i) != -1) {
166 					content += "°";
167 				} 
168 				content += "<br>";
169 			});
170 			content += "<strong>Source:</strong> <a href='" + this.sourceUrl + "' class='event-url' target='_blank'>" + this.sourceUrl + "</a><br></div>";
171 		
172 		var t = new Tip(this.container, content, {
173 			title: 'Details:',
174 			style: 'protogrey',
175 			stem: 'topLeft',
176 			closeButton: true,
177 			showOn: 'click',
178 			hideOn: 'click',
179 			hook: { target: 'bottomRight', tip: 'topLeft' },
180 			stem: 'topLeft',
181 			offset: { x: 14, y: 14 }
182 		});
